      : ps(new std::string(s)), i(0) {}
  HasPtr3(const HasPtr3& hp) : ps(new std::string(*hp.ps)), i(hp.i) {
#ifndef NDEBUG
    std::cout << "copied" << std::endl;
#endif
  }
  HasPtr3(HasPtr3&& hp) noexcept : ps(hp.ps), i(hp.i) {
    hp.ps = nullptr;
#ifndef NDEBUG
    std::cout << "moved" << std::endl;
#endif
  }

  // = 4s
  // mixed 4.8s

    auto newp = new std::string(*rhs.ps);
    delete ps;
    ps = newp;
    i = rhs.i;
#ifndef NDEBUG

#endif
    return *this;

#endif
